\vspace{2mm} {\small 适用场景：新车交付 / 二手车提车参考 | 发布日期：2026-01-17} \end{center} \section{1. 核心信息与证件核对 (身份确认)} \begin{itemize}[leftmargin=1.5em] \item \textbf{车架号 (VIN) 核对：} 与合同、合格证、发票、前挡玻璃左下角、B柱铭牌五处一致。 \item \textbf{生产日期：} 建议 $\le 3$ 个月（超过6个月需确认是否为库存车）。 \item \textbf{里程表读数：} 理想状态 $\le 20$km，超过 50km 需重点询问原因。 \item \textbf{钥匙核对：} 至少 2 把智能/机械钥匙，功能操作正常。 \end{itemize} \section{2. 外观与玻璃/轮胎日期} \begin{itemize} \item \textbf{车漆检查：} 逆光观察，检查前后保险杠、车门边缘是否有色差、划痕或补漆痕迹。 \item \textbf{接缝密合度：} 引擎盖、后备箱、四个车门缝隙是否均匀对称。 \item \textbf{全车玻璃日期：} 检查日期代码，应早于整车出厂日期且全车基本一致。 \item \textbf{轮胎生产日期：} 检查 4 位数字编码（如 3523 表示23年35周），四条胎应接近且早于整车日期。 \item \textbf{车灯外观：} 检查灯罩内是否有水雾、裂纹或严重划痕。 \end{itemize} \section{3. 内饰、功能与电子设备} \begin{itemize} \item \textbf{内饰完整性：} 拆开塑料膜检查座椅皮质是否有褶皱、污渍；安全带拉伸顺畅。 \item \textbf{空调系统：} 测试冷/暖风，检查最大风量噪音及出风口异味。 \item \textbf{中控/液晶屏：} 检查屏幕坏点、触摸灵敏度、CarPlay/蓝牙连接。 \item \textbf{按键/天窗：} 喇叭、车窗升降（防夹）、天窗开合无异响、遮阳帘正常。 \item \textbf{辅助影像：} 倒车影像/360环影清晰度，雷达报警声及引导线随动。 \end{itemize} \section{4. 机械试车 (建议园区内路试)} \begin{itemize} \item \textbf{发动机启动：} 冷车启动顺畅，怠速平稳无异响。 \item \textbf{换挡感受：} P-R-N-D 切换顺滑，无明显顿挫或金属撞击声。 \item \textbf{制动/转向：} 刹车灵敏无异响，方向盘回正自然，无卡滞。 \end{itemize} \section{5. 交付文件与赠品 (离店前必查)} \begin{itemize} \item 发票（三联）、合格证、一致性证书、环保清单、三包凭证、说明书、绿本（全款）。 \item 随车工具：三角牌、反光背心、灭火器、备胎/补胎工具、千斤顶。 \item 赠品核对：行车记录仪（含卡）、车膜（质保卡）、脚垫、保养次数（入系统）。 \end{itemize} \vspace{1cm} \begin{tabularx}{\textwidth}{l X
